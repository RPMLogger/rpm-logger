const SHEET_ID = '1GJB4BGETT4zeG1M7AKk48rhLaQw5ZqlImIDBC4ZepW4'; function doGet(e) { var action = e.parameter.action; if (action === 'getTodayStudents') return ContentService.createTextOutput(JSON.stringify(getTodayStudents())).setMimeType(ContentService.MimeType.JSON); if (action === 'getAllStudents') return ContentService.createTextOutput(JSON.stringify(getAllStudents())).setMimeType(ContentService.MimeType.JSON); if (action === 'logLesson') return ContentService.createTextOutput(JSON.stringify(logLesson(e.parameter))).setMimeType(ContentService.MimeType.JSON); if (action === 'logTrial') return ContentService.createTextOutput(JSON.stringify(logTrial(e.parameter))).setMimeType(ContentService.MimeType.JSON); if (action === 'logPaymentNote') return ContentService.createTextOutput(JSON.stringify(logPaymentNote(e.parameter))).setMimeType(ContentService.MimeType.JSON); if (action === 'getPastLessons') return ContentService.createTextOutput(JSON.stringify(getPastLessons(e.parameter))).setMimeType(ContentService.MimeType.JSON); return ContentService.createTextOutput(JSON.stringify({ error: 'Unknown action' })).setMimeType(ContentService.MimeType.JSON); } function getTodayStudents() { try { var today = new Date(); var startOfDay = new Date(today.getFullYear(), today.getMonth(), today.getDate(), 0, 0, 0); var endOfDay = new Date(today.getFullYear(), today.getMonth(), today.getDate(), 23, 59, 59); var calendars = CalendarApp.getAllCalendars(); var students = []; calendars.forEach(function(cal) { var events = cal.getEvents(startOfDay, endOfDay); var calName = cal.getName(); var isTrial = calName === 'RPM - Trial'; events.forEach(function(event) { var title = event.getTitle(); students.push({ name: title, trial: isTrial }); }); }); return { success: true, students: students }; } catch (err) { return { success: false, message: err.toString() }; } } function getAllStudents() { try { var ss = SpreadsheetApp.openById(SHEET_ID); var sheets = ss.getSheets(); var excluded = ['BLANK', 'LOAD', 'APPLICANTS', 'ZAM', 'FINANCIAL', 'CON', 'CONCERT', 'EXPENSES', 'REVIEW']; var students = []; sheets.forEach(function(sheet) { var name = sheet.getName(); if (excluded.indexOf(name) === -1 && name.indexOf('---') === -1) { students.push(name); } }); return { success: true, students: students }; } catch (err) { return { success: false, message: err.toString() }; } } function logLesson(params) { try { var studentName = params.studentName; var subject = params.subject; var firstName = studentName.split(' ')[0]; var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = findSheet(ss, firstName); if (!sheet) return { success: false, message: 'No sheet found for ' + studentName }; var targetRow = findNextLessonRow(sheet); var today = new Date(); var monthName = today.toLocaleString('en-US', { month: 'short' }); var day = today.getDate(); var dateStr = monthName + ' / ' + day; sheet.getRange('B' + targetRow + ':H' + targetRow).merge().setValue(subject).setHorizontalAlignment('right'); sheet.getRange('I' + targetRow).setValue(dateStr); return { success: true, message: 'Lesson logged for ' + studentName }; } catch (err) { return { success: false, message: err.toString() }; } } function logTrial(params) { try { var studentName = params.studentName; var subject = params.subject; var paid = params.trialPaid === '1'; var firstName = studentName.split(' ')[0].toUpperCase(); var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = findSheet(ss, firstName); if (!sheet) { var blankSheet = ss.getSheetByName('---BLANK---'); if (!blankSheet) return { success: false, message: 'No BLANK template found' }; blankSheet.copyTo(ss).setName(firstName); sheet = ss.getSheetByName(firstName); sheet.getRange('D2:J2').merge().setValue(studentName); sortSheetsAlphabetically(); } var today = new Date(); var monthName = today.toLocaleString('en-US', { month: 'short' }); var day = today.getDate(); var dateStr = monthName + ' / ' + day; sheet.getRange('D9:H9').merge().setValue(subject); var i9 = sheet.getRange('I9'); i9.setNumberFormat('@STRING@'); i9.setValue(dateStr); if (paid) sheet.getRange('J9').setValue(true); return { success: true, message: 'Trial logged for ' + studentName }; } catch (err) { return { success: false, message: err.toString() }; } } function logPaymentNote(params) { try { var studentName = params.studentName; var note = params.note; var firstName = studentName.split(' ')[0]; var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = findSheet(ss, firstName); if (!sheet) return { success: false, message: 'No sheet found for ' + studentName }; // Scan J12:J300 for first unchecked checkbox (false, not empty) var data = sheet.getRange('J12:J300').getValues(); var targetRow = -1; for (var i = 0; i < data.length; i++) { if (data[i][0] === false) { targetRow = 12 + i; break; } } if (targetRow === -1) return { success: false, message: 'No unchecked payment row found for ' + studentName }; // Check the checkbox sheet.getRange('J' + targetRow).setValue(true); // Find the actual top of the K merged cell using getMergedRanges if (note && note.trim() !== '') { var kCell = sheet.getRange('K' + targetRow); var kMerges = kCell.getMergedRanges(); var kTopRow = targetRow; if (kMerges.length > 0) { kTopRow = kMerges[0].getRow(); } sheet.getRange('K' + kTopRow).setValue(note); } return { success: true, message: 'Payment logged for ' + studentName + ' row ' + targetRow }; } catch (err) { return { success: false, message: err.toString() }; } } function getPastLessons(params) { try { var studentName = params.studentName; var count = parseInt(params.count) || 4; var firstName = studentName.split(' ')[0]; var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = findSheet(ss, firstName); if (!sheet) return { success: false, message: 'No sheet found for ' + studentName }; // Read rows 12 onwards — columns B:K // B:H = subject (merged), I = date, J = payment checkbox, K = payment note var data = sheet.getRange('B12:K300').getValues(); var lessons = []; for (var i = 0; i < data.length; i++) { var subject = data[i][0]; // Column B (merged B:H, value lands in B) var date = data[i][7]; // Column I var paid = data[i][8]; // Column J (checkbox) var note = data[i][9]; // Column K // Only rows with a date are real lesson entries if (date && date.toString().trim() !== '') { lessons.push({ subject: subject ? subject.toString().trim() : '', date: date.toString().trim(), paid: paid === true || paid === 'TRUE', paymentNote: note ? note.toString().trim() : '' }); } } // Return last N lessons, newest first var recent = lessons.slice(-count).reverse(); return { success: true, lessons: recent }; } catch (err) { return { success: false, message: err.toString() }; } } function findSheet(ss, studentName) { var sheets = ss.getSheets(); var firstName = studentName.split(' ')[0].toUpperCase(); for (var i = 0; i < sheets.length; i++) { if (sheets[i].getName().toUpperCase() === firstName) { return sheets[i]; } } return null; } function findNextLessonRow(sheet) { var data = sheet.getRange('I12:I300').getValues(); var lastRow = 11; for (var i = 0; i < data.length; i++) { if (data[i][0] !== '' && data[i][0] !== null) { lastRow = 12 + i; } } return lastRow + 1; } function sortSheetsAlphabetically() { var ss = SpreadsheetApp.openById(SHEET_ID); var sheets = ss.getSheets(); sheets.sort(function(a, b) { return a.getName().localeCompare(b.getName()); }); sheets.forEach(function(sheet, i) { ss.moveActiveSheet(i + 1); sheet.activate(); }); } function testTerry() { var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = findSheet(ss, 'Terry'); Logger.log(sheet ? sheet.getName() : 'NOT FOUND');
